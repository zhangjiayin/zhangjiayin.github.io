SELECT DISTINCT date( created\_at ) AS 'est\_date'  
FROM `clip`   
WHERE `created_at` \> DATE\_SUB( NOW( ) , INTERVAL DAYOFMONTH( NOW( ) )   
DAY )   
AND `channel_id` = 'xxxx'  
LIMIT 0 , 30
